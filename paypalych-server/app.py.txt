@app.route("/paypalych/result", methods=["POST"])
def paypalych_result():
    """Обработка успешного/неуспешного платежа"""
    
    # PayPalych может присылать данные в form-data или JSON
    if request.is_json:
        data = request.get_json()
    else:
        data = request.form.to_dict()
    
    logger.info(f"📥 Result URL: {json.dumps(data, indent=2, ensure_ascii=False)}")
    
    # Основные поля (названия зависят от PayPalych, это пример)
    order_id = data.get("order_id") or data.get("OrderId") or data.get("order")
    amount = data.get("amount") or data.get("Amount") or data.get("sum")
    currency = data.get("currency") or data.get("Currency") or "RUB"
    status = data.get("status") or data.get("Status") or data.get("payment_status")
    payment_method = data.get("method") or data.get("PaymentMethod") or data.get("type")
    customer_id = data.get("customer_id") or data.get("CustomerId") or data.get("user_id")
    signature = data.get("signature") or data.get("Signature") or data.get("hash")
    
    # Проверка подписи
    if signature and not verify_signature(data, signature):
        logger.error(f"❌ Неверная подпись для заказа #{order_id}")
        return "Invalid signature", 403
    
    if status in ["success", "paid", "completed", "1", "ok"]:
        # ✅ Платёж успешен
        logger.info(f"✅ Заказ #{order_id} оплачен: {amount} {currency}")
        
        # Тут будет начисление звёзд пользователю
        # await add_stars_to_user(customer_id, amount)
        
        # Можно отправить уведомление в Telegram
        # await bot.send_message(chat_id, f"✅ Оплата #{order_id} на {amount}₽ прошла!")
        
        return "OK", 200
    
    elif status in ["fail", "failed", "error", "0", "cancelled"]:
        # ❌ Платёж не прошёл
        logger.warning(f"❌ Заказ #{order_id} не оплачен: {status}")
        return "OK", 200  # Всё равно отвечаем 200, чтобы PayPalych не повторял запрос
    
    else:
        # Неизвестный статус — логируем для разбора
        logger.warning(f"⚠️ Неизвестный статус для заказа #{order_id}: {status}")
        return "OK", 200
